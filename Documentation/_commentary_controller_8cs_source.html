294 private void OnCommentaryComplete(){

296 if(CommentaryOn){
297 switch(currentCommentaryClip){
298 case CommentaryClip.PaintingIntroduction:
299 currentCommentaryClip = CommentaryClip.CommentaryOffTest;
300 break;
301 case CommentaryClip.CommentaryOffTest:
302 currentCommentaryClip = CommentaryClip.Intro;
303 PlayClip(currentLocation.ToString() + CommentaryClip.Intro.ToString());
304 break;
305 case CommentaryClip.Intro:
306 currentCommentaryClip = CommentaryClip.MoreArtist;
307 waitingForNodOrShake = true;
308 PlayClip(currentLocation.ToString() + CommentaryClip.MoreArtist.ToString());
309 break;
310 case CommentaryClip.Artist:
311 currentCommentaryClip = CommentaryClip.MorePainting;
312 waitingForNodOrShake = true;
313 PlayClip(currentLocation.ToString() + CommentaryClip.MorePainting.ToString());
314 break;
315 case CommentaryClip.Painting:
316 currentCommentaryClip = CommentaryClip.MoreMusic;
317 waitingForNodOrShake = true;
318 PlayClip(currentLocation.ToString() + CommentaryClip.MoreMusic.ToString());
319 break;
320 case CommentaryClip.Music:
321 currentCommentaryClip = CommentaryClip.MoreArtist;
322 waitingForNodOrShake = true;
323 PlayClip(currentLocation.ToString() + CommentaryClip.MoreArtist.ToString());
324 break;
325 default:
326 FMODUnity.RuntimeManager.StudioSystem.setParameterByName("DuckMusic", 0);
327 break;
328 }

331 FMODUnity.RuntimeManager.StudioSystem.setParameterByName("DuckMusic", 0);

351 private void PlayClip(string commentaryClip){
352 if(commentaryTalking){
353 commentaryClipInstance.stop(FMOD.Studio.STOP_MODE.ALLOWFADEOUT);
354 }
355 commentaryClipInstance = FMODUnity.RuntimeManager.CreateInstance("event:/Commentary/" + commentaryClip);
356 if(audioGuide){
357 FMODUnity.RuntimeManager.StudioSystem.setParameterByName("DuckMusic", 40);
358 FMODUnity.RuntimeManager.AttachInstanceToGameObject(commentaryClipInstance, audioGuide.GetComponent<Transform>(), audioGuide.GetComponent<Rigidbody>());
359 commentaryTalking = true;
360 commentaryClipInstance.start();
361 }
